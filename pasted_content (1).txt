    container.innerHTML = `<div class="sim-inner"><h3>🧮 Explorador de Valor Posicional</h3><p>Digite os dígitos para decompor o número:</p><div class="explorador-wrap"><div class="explorador-inputs" id="explInputs"><div class="explorador-col"><label>Milhar</label><input type="number" id="exMil" min="0" max="9" value="2" oninput="atualizarExplorador()"></div><div class="explorador-sep">.</div><div class="explorador-col"><label>Centena</label><input type="number" id="exCen" min="0" max="9" value="4" oninput="atualizarExplorador()"></div><div class="explorador-sep">.</div><div class="explorador-col"><label>Dezena</label><input type="number" id="exDez" min="0" max="9" value="5" oninput="atualizarExplorador()"></div><div class="explorador-sep">.</div><div class="explorador-col"><label>Unidade</label><input type="number" id="exUni" min="0" max="9" value="7" oninput="atualizarExplorador()"></div></div><div class="explorador-result-box" id="explResult"></div><div class="explorador-blocks" id="explBlocks" style="margin-top:0.6rem;min-height:70px;"></div></div></div>`;

function atualizarExplorador() {
    const m=Math.max(0,Math.min(9,parseInt(document.getElementById('exMil')?.value)||0));
    const c=Math.max(0,Math.min(9,parseInt(document.getElementById('exCen')?.value)||0));
    const d=Math.max(0,Math.min(9,parseInt(document.getElementById('exDez')?.value)||0));
    const u=Math.max(0,Math.min(9,parseInt(document.getElementById('exUni')?.value)||0));
    const total=m*1000+c*100+d*10+u;
    const fmt=total.toLocaleString('pt-BR');
    const resultBox=document.getElementById('explResult');
    if(!resultBox)return;
    let dp=[];
    if(m)dp.push(`<span style="color:var(--secondary)">${m}×1000</span>`);
    if(c)dp.push(`<span style="color:var(--warning)">${c}×100</span>`);
    if(d)dp.push(`<span style="color:var(--info)">${d}×10</span>`);
    if(u)dp.push(`<span style="color:var(--success)">${u}×1</span>`);
    if(!dp.length)dp.push('<span style="color:var(--text-tertiary)">0</span>');
    resultBox.innerHTML=`<div class="explorador-num">${fmt}</div><div class="explorador-breakdown">${dp.join(' + ')} = <strong style="color:var(--primary)">${fmt}</strong></div>`;
    const bl=document.getElementById('explBlocks');
    if(!bl)return;
    let h='';
    for(let i=0;i<m;i++)h+=`<div class="block-mil"></div>`;
    for(let i=0;i<c;i++)h+=`<div class="block-cent"></div>`;
    for(let i=0;i<d;i++)h+=`<div class="block-dez"></div>`;
    for(let i=0;i<u;i++)h+=`<div class="block-unit"></div>`;
    if(!h)h=`<span style="font-size:0.75rem;color:var(--text-tertiary)">Digite dígitos acima.</span>`;
    bl.innerHTML=h;
}

/* ── RETA NUMÉRICA — ponto arrastável, sincroniza alternativas da atividade ── */
simulators.reta = function(c) {
    c.innerHTML = `<div class="sim-inner">
        <h3>📏 Reta Numérica Interativa</h3>
        <p style="font-size:0.78rem;color:var(--text-secondary);margin-bottom:6px;">Arraste o ponto laranja ou use o controle abaixo.</p>
        ${_simCanvas('retaCV',80)}
        ${_simSlider('retaSlider',-10,10,2,1,'_retaUpdate()')}
        <div id="retaInfo" class="explicacao-box" style="margin-top:6px;"></div>
        <div id="retaOpts" style="display:flex;gap:0.4rem;flex-wrap:wrap;margin-top:8px;"></div>
    </div>`;
    setTimeout(()=>{
        const cv=document.getElementById('retaCV');
        if(!cv)return;
        cv.width=cv.offsetWidth||400; cv.height=80;
        let pos=2;
        const W=cv.width, CX=W/2, STEP=W/22;
        function draw(){
            const ctx=cv.getContext('2d');
            ctx.fillStyle='#04060D'; ctx.fillRect(0,0,W,80);
            // linha
            ctx.strokeStyle='#4F8EF7'; ctx.lineWidth=2;
            ctx.beginPath(); ctx.moveTo(12,40); ctx.lineTo(W-12,40); ctx.stroke();
            // marcas
            for(let i=-10;i<=10;i++){
                const x=CX+i*STEP;
                ctx.beginPath(); ctx.moveTo(x,33); ctx.lineTo(x,47);
                ctx.strokeStyle=i===0?'rgba(255,255,255,0.6)':'rgba(255,255,255,0.25)';
                ctx.lineWidth=i===0?2:1; ctx.stroke();
                ctx.fillStyle='#7A8699'; ctx.font='9px JetBrains Mono';
                ctx.fillText(i,x-(i<0?5:3),62);
            }
            // ponto
            const px=CX+pos*STEP;
            ctx.beginPath(); ctx.arc(px,40,8,0,Math.PI*2);
            ctx.fillStyle='rgba(245,166,35,0.25)'; ctx.fill();
            ctx.beginPath(); ctx.arc(px,40,5,0,Math.PI*2);
            ctx.fillStyle='#F5A623'; ctx.fill();
            ctx.strokeStyle='#fff'; ctx.lineWidth=1.5; ctx.stroke();
        }
        function sync(){
            document.getElementById('retaSlider').value=pos;
            document.getElementById('retaInfo').innerHTML=
                `O ponto laranja está na posição <strong style="color:#F5A623">${pos}</strong>. `+
                (pos>0?`Número inteiro positivo.`:pos<0?`Número inteiro negativo.`:`É o zero, origem da reta.`);
            _retaHighlightOpts(pos);
            draw();
        }
        // drag mouse
        let drag=false;
        cv.addEventListener('mousedown',()=>drag=true);
        window.addEventListener('mouseup',()=>drag=false);
        cv.addEventListener('mousemove',e=>{
            if(!drag)return;
            const r=cv.getBoundingClientRect();
            const ratio=cv.width/r.width;
            const mx=(e.clientX-r.left)*ratio;
            pos=Math.round((mx-CX)/STEP);
            pos=Math.max(-10,Math.min(10,pos));
            sync();
        });
        // drag touch
        cv.addEventListener('touchstart',e=>{drag=true;e.preventDefault();},{passive:false});
        cv.addEventListener('touchend',()=>drag=false);
        cv.addEventListener('touchmove',e=>{
            if(!drag)return; e.preventDefault();
            const r=cv.getBoundingClientRect();
            const ratio=cv.width/r.width;
            const mx=(e.touches[0].clientX-r.left)*ratio;
            pos=Math.round((mx-CX)/STEP);
            pos=Math.max(-10,Math.min(10,pos));
            sync();
        },{passive:false});
        window._retaUpdate=function(){
            pos=parseInt(document.getElementById('retaSlider').value);
            sync();
        };
        window._retaHighlightOpts=function(v){
            const box=document.getElementById('retaOpts');
            if(!box)return;
            const opts=_getActiveSimOpts();
            if(!opts.length){box.innerHTML='';return;}
            box.innerHTML='<span style="font-size:0.72rem;color:var(--text-secondary);">Alternativas da questão:</span>'
                +opts.map((o,i)=>{
                    const n=parseInt(o.replace(/[^-\d]/g,''));
                    const match=!isNaN(n)&&n===v;
                    return `<span onclick="_retaJump(${n||i})" style="padding:3px 10px;border-radius:20px;cursor:pointer;font-size:0.8rem;font-weight:700;
                        background:${match?'var(--warning)':'var(--surface-3)'};color:${match?'#000':'var(--text)'};border:1px solid ${match?'var(--warning)':'var(--border)'};transition:0.2s;">${o}</span>`;
                }).join('');
        };
        window._retaJump=function(v){
            pos=Math.max(-10,Math.min(10,isNaN(v)?0:v));
            sync();
        };
        sync();
    },80);
};

simulators.trigonometria = function(c) {
    c.innerHTML = `<div class="sim-inner">
        <h3>📐 Trigonometria — Triângulo Retângulo</h3>
        <label style="font-size:0.7rem;color:var(--text-secondary);">Ângulo α (graus)</label>
        ${_simSlider('trigA',1,89,30,1,'_trigCalc()')}
        <div id="trigAv" style="text-align:center;font-weight:700;color:var(--warning);font-size:1.1rem;">30°</div>
        ${_simCanvas('trigCV',140)}
        <div id="trigInfo" class="explicacao-box" style="margin-top:6px;"></div>
        <div id="trigOpts" style="display:flex;gap:0.4rem;flex-wrap:wrap;margin-top:8px;"></div>
    </div>`;
    setTimeout(()=>{
        const cv=document.getElementById('trigCV');
        if(!cv)return;
        cv.width=cv.offsetWidth||380; cv.height=140;
        window._trigCalc=function(){
            const deg=parseInt(document.getElementById('trigA').value);
            document.getElementById('trigAv').textContent=deg+'°';
            const rad=deg*Math.PI/180;
            const sen=Math.sin(rad), cos=Math.cos(rad), tg=Math.tan(rad);
            const ctx=cv.getContext('2d');
            const W=cv.width,H=140;
            ctx.fillStyle='#04060D'; ctx.fillRect(0,0,W,H);
            const L=100, ox=40, oy=H-30;
            const ax=ox+L*cos, ay=oy-L*sen;
            ctx.beginPath(); ctx.moveTo(ox,oy); ctx.lineTo(ox+L*cos,oy); ctx.lineTo(ax,ay); ctx.closePath();
            ctx.fillStyle='rgba(79,142,247,0.1)'; ctx.fill();
            ctx.strokeStyle='rgba(79,142,247,0.7)'; ctx.lineWidth=2; ctx.stroke();
            // ângulo reto
            ctx.beginPath(); ctx.moveTo(ox+L*cos-10,oy); ctx.lineTo(ox+L*cos-10,oy-10); ctx.lineTo(ox+L*cos,oy-10);
            ctx.strokeStyle='rgba(255,255,255,0.4)'; ctx.lineWidth=1; ctx.stroke();
            // arco do ângulo α
            ctx.beginPath(); ctx.arc(ox,oy,22,0,-rad,true);
            ctx.strokeStyle='#F5A623'; ctx.lineWidth=2; ctx.stroke();
            ctx.fillStyle='#F5A623'; ctx.font='bold 11px Space Grotesk';
            ctx.fillText(`α=${deg}°`,ox+24,oy-4);
            // labels lados
            ctx.fillStyle='#22D3EE'; ctx.fillText(`adj=${+(L*cos).toFixed(1)}`,ox+L*cos/2-10,oy+14);
            ctx.fillStyle='#F5A623'; ctx.fillText(`op=${+(L*sen).toFixed(1)}`,ox+L*cos+6,(oy+ay)/2);
            ctx.fillStyle='#22C97A'; ctx.fillText(`hip=${L}`,(ox+ax)/2-18,(oy+ay)/2-8);
            document.getElementById('trigInfo').innerHTML=
                `sen(${deg}°) = <strong style="color:var(--success)">${+sen.toFixed(4)}</strong> | `+
                `cos(${deg}°) = <strong style="color:var(--info)">${+cos.toFixed(4)}</strong> | `+
                `tg(${deg}°) = <strong style="color:var(--warning)">${+tg.toFixed(4)}</strong>`;
            _trigHighlight(+sen.toFixed(4),+cos.toFixed(4),+tg.toFixed(4));
        };
        window._trigHighlight=function(s,co,t){
            const box=document.getElementById('trigOpts');
            if(!box)return;
            const opts=_getActiveSimOpts();
            if(!opts.length){box.innerHTML='';return;}
            box.innerHTML='<span style="font-size:0.72rem;color:var(--text-secondary);">Alternativas:</span>'
                +opts.map(o=>{
                    const n=parseFloat(o.replace(',','.').replace(/[^-\d.]/g,''));
                    const match=!isNaN(n)&&(Math.abs(n-s)<0.01||Math.abs(n-co)<0.01||Math.abs(n-t)<0.01);
                    return `<span style="padding:3px 10px;border-radius:20px;font-size:0.8rem;font-weight:700;
                        background:${match?'var(--warning)':'var(--surface-3)'};color:${match?'#000':'var(--text)'};border:1px solid ${match?'var(--warning)':'var(--border)'};transition:0.2s;">${o}</span>`;
                }).join('');
        };
        window._trigCalc();
    },80);
};

/* ── FUNÇÃO AFIM — slider de a e b ── */
simulators.afim = function(c) {
    c.innerHTML = `<div class="sim-inner">
        <h3>📈 Função Afim f(x)=ax+b</h3>
        <div style="display:grid;grid-template-columns:1fr 1fr;gap:10px;margin:6px 0;">
            <div><label style="font-size:0.7rem;color:var(--text-secondary);">Coef. angular (a)</label>
            ${_simSlider('afimA',-5,5,2,0.5,'_afimCalc()')}<div id="afimAv" style="text-align:center;font-weight:700;color:var(--warning);">2</div></div>
            <div><label style="font-size:0.7rem;color:var(--text-secondary);">Coef. linear (b)</label>
            ${_simSlider('afimB',-8,8,3,1,'_afimCalc()')}<div id="afimBv" style="text-align:center;font-weight:700;color:var(--info);">3</div></div>
        </div>
        <div style="font-size:0.72rem;color:var(--text-secondary);margin-bottom:4px;">Calcule f(x) para:</div>
        <div style="display:flex;gap:8px;align-items:center;margin-bottom:8px;">
            <input id="afimX" type="number" value="1" style="width:60px;background:var(--surface-3);border:1px solid var(--border-2);border-radius:8px;padding:6px;color:var(--text);font-weight:700;text-align:center;" oninput="_afimCalc()">
            <span style="color:var(--text-secondary);">→ f(x) = </span>
            <strong id="afimFx" style="color:var(--primary);font-size:1.1rem;">7</strong>
        </div>
        ${_simCanvas('afimCV',140)}
        <div id="afimInfo" class="explicacao-box" style="margin-top:6px;"></div>
        <div id="afimOpts" style="display:flex;gap:0.4rem;flex-wrap:wrap;margin-top:8px;"></div>
    </div>`;
    setTimeout(()=>{
        const cv=document.getElementById('afimCV');
        if(!cv)return;
        cv.width=cv.offsetWidth||380; cv.height=140;
        window._afimCalc=function(){
            const a=parseFloat(document.getElementById('afimA').value)||0;
            const b=parseFloat(document.getElementById('afimB').value)||0;
            const xv=parseFloat(document.getElementById('afimX').value)||0;
            document.getElementById('afimAv').textContent=a;
            document.getElementById('afimBv').textContent=b;
            const fx=a*xv+b;
            document.getElementById('afimFx').textContent=+fx.toFixed(3);
            const ctx=cv.getContext('2d');
            const W=cv.width,H=140,ox=W/2,oy=H/2,scX=22,scY=12;
            ctx.fillStyle='#04060D'; ctx.fillRect(0,0,W,H);
            ctx.strokeStyle='rgba(255,255,255,0.15)'; ctx.lineWidth=1;
            ctx.beginPath(); ctx.moveTo(0,oy); ctx.lineTo(W,oy); ctx.stroke();
            ctx.beginPath(); ctx.moveTo(ox,0); ctx.lineTo(ox,H); ctx.stroke();
            ctx.strokeStyle='#4F8EF7'; ctx.lineWidth=2.5;
            ctx.beginPath();
            for(let px=0;px<=W;px++){
                const x=(px-ox)/scX;
                const y=a*x+b;
                const py=oy-y*scY;
                px===0?ctx.moveTo(px,py):ctx.lineTo(px,py);
            }
            ctx.stroke();
            // ponto f(xv)
            const ppx=ox+xv*scX, ppy=oy-fx*scY;
            if(ppy>=0&&ppy<=H){
                ctx.beginPath(); ctx.arc(ppx,ppy,6,0,Math.PI*2);
                ctx.fillStyle='#F5A623'; ctx.fill();
                ctx.strokeStyle='#fff'; ctx.lineWidth=1.5; ctx.stroke();
            }
            const bStr=b>=0?`+${b}`:b;
            document.getElementById('afimInfo').innerHTML=
                `f(x)=${a}x${bStr} | f(${xv}) = ${a}·${xv}${bStr} = <strong style="color:var(--primary)">${+fx.toFixed(3)}</strong>`+
                (a>0?' (crescente)':`${a<0?' (decrescente)':' (constante)'}`);
            _afimHighlight(+fx.toFixed(3));
        };
        window._afimHighlight=function(fx){
            const box=document.getElementById('afimOpts');
            if(!box)return;
            const opts=_getActiveSimOpts();
            if(!opts.length){box.innerHTML='';return;}
            box.innerHTML='<span style="font-size:0.72rem;color:var(--text-secondary);">Alternativas:</span>'
                +opts.map(o=>{
                    const n=parseFloat(o.replace(',','.').replace(/[^-\d.]/g,''));
                    const match=!isNaN(n)&&Math.abs(n-fx)<0.1;
                    return `<span style="padding:3px 10px;border-radius:20px;font-size:0.8rem;font-weight:700;
                        background:${match?'var(--primary)':'var(--surface-3)'};color:${match?'#fff':'var(--text)'};border:1px solid ${match?'var(--primary)':'var(--border)'};transition:0.2s;">${o}</span>`;
                }).join('');
        };
        window._afimCalc();
    },80);
};

        if(!cv)return;
        cv.width=cv.offsetWidth||380; cv.height=140;

            const xv=parseFloat(document.getElementById('expX').value)||0;
            document.getElementById('expAv').textContent=+a.toFixed(1);
            const fx=Math.pow(a,xv);
            document.getElementById('expFx').textContent=+fx.toFixed(4);
            const ctx=cv.getContext('2d');
            const W=cv.width,H=140,ox=W/3,oy=H-20,scX=25,scY=2.5;

            for(let px=0;px<=W;px++){

/* ── CICLO TRIGONOMÉTRICO — círculo unitário interativo ── */
simulators.ciclo = function(c) {
    c.innerHTML = `<div class="sim-inner">
        <h3>⭕ Ciclo Trigonométrico</h3>
        <label style="font-size:0.7rem;color:var(--text-secondary);">Ângulo (graus)</label>
        ${_simSlider('cicA',0,360,30,1,'_cicCalc()')}
        <div id="cicAv" style="text-align:center;font-weight:700;color:var(--warning);font-size:1.1rem;">30°</div>
        ${_simCanvas('cicCV',160)}
        <div id="cicInfo" class="explicacao-box" style="margin-top:6px;"></div>
        <div id="cicOpts" style="display:flex;gap:0.4rem;flex-wrap:wrap;margin-top:8px;"></div>
    </div>`;
    setTimeout(()=>{
        const cv=document.getElementById('cicCV');
        if(!cv)return;
        cv.width=cv.offsetWidth||380; cv.height=160;
        window._cicCalc=function(){
            const deg=parseInt(document.getElementById('cicA').value);
            document.getElementById('cicAv').textContent=deg+'°';
            const rad=deg*Math.PI/180;
            const s=Math.sin(rad),co=Math.cos(rad);
            const ctx=cv.getContext('2d');
            const W=cv.width,H=160,R=62,CX=W/2,CY=H/2+5;
            ctx.fillStyle='#04060D'; ctx.fillRect(0,0,W,H);
            // eixos
            ctx.strokeStyle='rgba(255,255,255,0.12)'; ctx.lineWidth=1;
            ctx.beginPath(); ctx.moveTo(CX-R-10,CY); ctx.lineTo(CX+R+10,CY); ctx.stroke();
            ctx.beginPath(); ctx.moveTo(CX,CY-R-10); ctx.lineTo(CX,CY+R+10); ctx.stroke();
            // círculo
            ctx.beginPath(); ctx.arc(CX,CY,R,0,2*Math.PI);
            ctx.strokeStyle='rgba(79,142,247,0.4)'; ctx.lineWidth=1.5; ctx.stroke();
            // raio
            const px=CX+R*co, py=CY-R*s;
            ctx.beginPath(); ctx.moveTo(CX,CY); ctx.lineTo(px,py);
            ctx.strokeStyle='#F5A623'; ctx.lineWidth=2.5; ctx.stroke();
            // ponto P
            ctx.beginPath(); ctx.arc(px,py,6,0,Math.PI*2);
            ctx.fillStyle='#F5A623'; ctx.fill();
            ctx.strokeStyle='#fff'; ctx.lineWidth=1.5; ctx.stroke();
            // projeções
            ctx.setLineDash([4,4]);
            ctx.beginPath(); ctx.moveTo(px,py); ctx.lineTo(px,CY);
            ctx.strokeStyle='#22C97A'; ctx.lineWidth=1.5; ctx.stroke();
            ctx.beginPath(); ctx.moveTo(px,py); ctx.lineTo(CX,py);
            ctx.strokeStyle='#22D3EE'; ctx.lineWidth=1.5; ctx.stroke();
            ctx.setLineDash([]);
            // labels
            ctx.fillStyle='#22C97A'; ctx.font='bold 10px Space Grotesk';
            ctx.fillText(`sen=${+s.toFixed(3)}`,CX+R+4,py);
            ctx.fillStyle='#22D3EE';
            ctx.fillText(`cos=${+co.toFixed(3)}`,px>CX?px-60:px+4,CY+14);
            document.getElementById('cicInfo').innerHTML=
                `P(${+co.toFixed(3)}, ${+s.toFixed(3)}) | `+
                `sen(${deg}°) = <strong style="color:var(--success)">${+s.toFixed(4)}</strong> | `+
                `cos(${deg}°) = <strong style="color:var(--info)">${+co.toFixed(4)}</strong>`;
            _cicHighlight(+s.toFixed(4),+co.toFixed(4));
        };
        window._cicHighlight=function(s,co){
            const box=document.getElementById('cicOpts');
            if(!box)return;
            const opts=_getActiveSimOpts();
            if(!opts.length){box.innerHTML='';return;}
            box.innerHTML='<span style="font-size:0.72rem;color:var(--text-secondary);">Alternativas:</span>'
                +opts.map(o=>{
                    const n=parseFloat(o.replace(',','.').replace(/[^-\d.]/g,''));
                    const match=!isNaN(n)&&(Math.abs(n-s)<0.01||Math.abs(n-co)<0.01);
                    return `<span style="padding:3px 10px;border-radius:20px;font-size:0.8rem;font-weight:700;
                        background:${match?'var(--secondary)':'var(--surface-3)'};color:${match?'#fff':'var(--text)'};border:1px solid ${match?'var(--secondary)':'var(--border)'};transition:0.2s;">${o}</span>`;
                }).join('');
        };
        window._cicCalc();
    },80);
};

/* ── PROGRESSÕES ── */
simulators.progressao = function(c) {
    c.innerHTML = `<div class="sim-inner">
        <h3>🔢 Progressões (PA e PG)</h3>
        <div style="display:flex;gap:10px;align-items:center;flex-wrap:wrap;margin:8px 0;">
            <select id="progTipo" style="background:var(--surface-2);border:1px solid var(--border-2);border-radius:8px;padding:6px 10px;color:var(--primary);font-weight:700;cursor:pointer;" onchange="_progCalc()">
                <option>PA</option><option>PG</option>
            </select>
            <div style="flex:1;min-width:100px;"><label style="font-size:0.65rem;color:var(--text-secondary);">1º Termo (a₁)</label>
            <input id="progA1" type="number" value="2" style="width:100%;background:var(--surface-3);border:1px solid var(--border-2);border-radius:8px;padding:6px;color:var(--text);font-weight:700;text-align:center;" oninput="_progCalc()"></div>
            <div style="flex:1;min-width:100px;"><label id="progRLabel" style="font-size:0.65rem;color:var(--text-secondary);">Razão r</label>
            <input id="progR" type="number" value="3" style="width:100%;background:var(--surface-3);border:1px solid var(--border-2);border-radius:8px;padding:6px;color:var(--text);font-weight:700;text-align:center;" oninput="_progCalc()"></div>
            <div style="flex:1;min-width:100px;"><label style="font-size:0.65rem;color:var(--text-secondary);">Termo nº n</label>
            <input id="progN" type="number" value="5" min="1" max="20" style="width:100%;background:var(--surface-3);border:1px solid var(--border-2);border-radius:8px;padding:6px;color:var(--text);font-weight:700;text-align:center;" oninput="_progCalc()"></div>
        </div>
        <div id="progSeq" style="background:var(--surface-3);border-radius:8px;padding:10px;font-family:'JetBrains Mono';font-size:0.85rem;overflow-x:auto;white-space:nowrap;margin-bottom:6px;"></div>
        <div id="progInfo" class="explicacao-box"></div>
        <div id="progOpts" style="display:flex;gap:0.4rem;flex-wrap:wrap;margin-top:8px;"></div>
    </div>`;
    window._progCalc=function(){
        const tipo=document.getElementById('progTipo').value;
        const a1=parseFloat(document.getElementById('progA1').value)||1;
        const r=parseFloat(document.getElementById('progR').value)||1;
        const n=Math.max(1,Math.min(20,parseInt(document.getElementById('progN').value)||5));
        document.getElementById('progRLabel').textContent=tipo==='PA'?'Razão r':'Razão q';
        const terms=Array.from({length:n},(_,i)=>tipo==='PA'?+(a1+i*r).toFixed(4):+(a1*Math.pow(r,i)).toFixed(4));
        document.getElementById('progSeq').innerHTML=
            terms.map((t,i)=>`<span style="color:${i===n-1?'var(--warning)':'var(--text-secondary)'}">${t}</span>`).join(` <span style="color:var(--primary);">→</span> `);
        const an=terms[n-1];
        document.getElementById('progInfo').innerHTML=
            tipo==='PA'
            ?`Fórmula: a${n} = ${a1} + (${n-1})×${r} = <strong style="color:var(--warning)">${an}</strong> | Sn = <strong style="color:var(--info)">${+(n*(a1+an)/2).toFixed(4)}</strong>`
            :`Fórmula: a${n} = ${a1} × ${r}^${n-1} = <strong style="color:var(--warning)">${an}</strong>`;
        _progHighlight(an,terms);
    };
    window._progHighlight=function(an,terms){
        const box=document.getElementById('progOpts');
        if(!box)return;
        const opts=_getActiveSimOpts();
        if(!opts.length){box.innerHTML='';return;}
        box.innerHTML='<span style="font-size:0.72rem;color:var(--text-secondary);">Alternativas:</span>'
            +opts.map(o=>{
                const n=parseFloat(o.replace(',','.').replace(/[^-\d.]/g,''));
                const match=!isNaN(n)&&(Math.abs(n-an)<0.1||terms.some(t=>Math.abs(t-n)<0.1));
                return `<span style="padding:3px 10px;border-radius:20px;font-size:0.8rem;font-weight:700;
                    background:${match?'var(--warning)':'var(--surface-3)'};color:${match?'#000':'var(--text)'};border:1px solid ${match?'var(--warning)':'var(--border)'};transition:0.2s;">${o}</span>`;
            }).join('');
    };
    window._progCalc();
};

/* ============================================================
   CHAMADA API COM RETRY ROBUSTO (HTTP 429)
   Correção principal: backoff exponencial + feedback ao usuário
   ============================================================ */
async function _chamarAPIComRetry(body, maxTentativas = 5) {
    // Delays progressivos: 1s, 2s, 4s, 8s, 15s (mais rápido)
    const delays = [1000, 2000, 4000, 8000, 15000];

    for (let tentativa = 0; tentativa < maxTentativas; tentativa++) {
        let response;
        try {
            // Timeout de 15s para evitar travamentos
            const controller = new AbortController();
            const timeoutId = setTimeout(() => controller.abort(), 15000);
            
            response = await fetch("https://api.anthropic.com/v1/messages", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify(body),
                signal: controller.signal
            });
            clearTimeout(timeoutId);
        } catch (netErr) {
            if (tentativa < maxTentativas - 1) {
                const espera = delays[tentativa];
                _mostrarStatusCarregamento(`⚠️ Conexão instável. Tentativa ${tentativa + 2}/${maxTentativas}…`);
                await _sleep(espera);
                continue;
            }
            throw new Error('Falha na comunicação com a IA');
        }

        if (response.ok) return response;

        if (response.status === 429 || response.status >= 500) {
            if (tentativa < maxTentativas - 1) {
                const espera = delays[tentativa];
                _mostrarStatusCarregamento(`⏳ Otimizando processamento... (${tentativa + 2}/${maxTentativas})`);
                await _sleep(espera);
                continue;
            }
        }
        throw new Error(`Erro API: ${response.status}`);
    }
    throw new Error('Máximo de tentativas atingido');
}

function _sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

function _mostrarStatusCarregamento(msg) {
    const mb = document.getElementById('atividadeModalBody');
    if (!mb) return;
    const statusEl = mb.querySelector('.loading-status-msg');
    if (statusEl) {
        statusEl.textContent = msg;
    } else {
        mb.innerHTML = `
            <div class="atividade-loading">
                <div class="loading-ring"></div>
                <div class="loading-dots">
                    <div class="loading-dot"></div>
                    <div class="loading-dot"></div>
                    <div class="loading-dot"></div>
                </div>
                <div class="loading-status-msg" style="font-size:0.85rem;color:var(--text-secondary);max-width:320px;text-align:center;">${msg}</div>
            </div>`;
    }
}

/* ============================================================
   LINK COMPARTILHÁVEL — URL funcional em qualquer navegador
   ============================================================ */
function gerarCodigoAtividade(tema, serie, tipoIdx) {
    const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
    let codigo = '';
    for (let i = 0; i < 8; i++) codigo += chars[Math.floor(Math.random() * chars.length)];
    return `${serie.toUpperCase()}-${tema.toUpperCase().slice(0, 4)}-T${tipoIdx}-${codigo}`;
}

function gerarLinkAtividade(codigo) {
    // Gera URL absoluta e limpa — funciona em qualquer navegador/contexto
    const proto = window.location.protocol;
    const host = window.location.host;
    // Remove parâmetros e hashes existentes do pathname
    const path = window.location.pathname.replace(/\/+$/, '') || '/';
    return `${proto}//${host}${path}?atividade=${encodeURIComponent(codigo)}`;
}

function copiarLink() {
    const input = document.getElementById('shareLinkInput');
    const val = input.value;
    if (!val) return;

    // Método moderno (HTTPS) com fallback para HTTP/contextos sem permissão
    if (navigator.clipboard && window.isSecureContext) {
        navigator.clipboard.writeText(val)
            .then(() => showToast('🔗 Link copiado!'))
            .catch(() => _copiarFallback(input));
    } else {
        _copiarFallback(input);
    }
}

function _copiarFallback(input) {
    input.removeAttribute('readonly');
    input.select();
    input.setSelectionRange(0, 99999);
    try {
        const ok = document.execCommand('copy');
        showToast(ok ? '🔗 Link copiado!' : '⚠️ Copie manualmente o link');
    } catch (e) {
        showToast('⚠️ Copie manualmente o link');
    }
    input.setAttribute('readonly', '');
}

async function carregarAtividade(tipoIdx) {
    const tipo = tiposAtividade[tipoIdx];
    const serieData = seriesData[atividadeState.serie] || seriesData.ef6;
    const cacheKey = `${atividadeState.tema}_${atividadeState.serie}_${tipoIdx}`;

    _mostrarStatusCarregamento(`Gerando atividade ${tipo.emoji} ${tipo.nome}… A IA está criando 10 questões exclusivas com imagens.`);

    if (atividadeState.cache[cacheKey]) {
        renderAtividade(atividadeState.cache[cacheKey], tipo);
        return;
    }

    const prompt = `Você é professor de matemática do Ensino Fundamental/Médio do estado de São Paulo (Currículo Paulista).

Gere EXATAMENTE 10 questões de múltipla escolha sobre o tema: "${atividadeState.tituloAE}" (${atividadeState.tema}).
Série/nível: ${serieData.name}.
Abordagem: ${tipo.abordagem}

REGRAS OBRIGATÓRIAS:
1. Cada questão tem EXATAMENTE 4 alternativas: A, B, C ou D
2. Apenas UMA alternativa é correta
3. Explicação detalhada da resposta correta (2-3 frases)
4. Dificuldade progressiva (1-3 fáceis, 4-7 médias, 8-10 difíceis)
5. Varie o tipo de raciocínio exigido
6. Para cada questão, sugira:
   - "imagemKeyword": palavra-chave em inglês (1-2 palavras) que descreve uma imagem visual para contextualizar o enunciado. Use: fraction, pizza, triangle, pythagorean, circle, parabola, graph, linear, exponential, decimal, clock, ruler, thermometer, bacteria, fibonacci, cartesian, sine, cube, sieve, population, money, multiplication
   - "imagemCaption": legenda curta em português (máx. 60 caracteres) descrevendo o contexto visual

Responda APENAS com JSON válido, sem markdown, sem explicações fora do JSON:
{"questoes":[{"enunciado":"texto","opcoes":["A. texto","B. texto","C. texto","D. texto"],"correta":0,"explicacao":"explicação","imagemKeyword":"keyword","imagemCaption":"legenda"}]}

"correta" = índice 0-3 da alternativa correta. Inclua a letra (A., B., C., D.) no início de cada opção.`;

    // Enfileira chamada para evitar 429 por paralelismo
    _apiQueue = _apiQueue.then(async () => {
        if (atividadeState.cache[cacheKey]) {
            renderAtividade(atividadeState.cache[cacheKey], tipo);
            return;
        }
        try {
            const response = await _chamarAPIComRetry({
                model: "gpt-4.1-mini", // Modelo mais rápido e eficiente para JSON
                max_tokens: 4000,
                system: "Você é professor especialista em matemática. Responda APENAS com JSON válido. Seja conciso e preciso.",
                messages: [{ role: "user", content: prompt }]
            });

            const data = await response.json();
            let raw = data.content
                .filter(b => b.type === 'text')
                .map(b => b.text || '')
                .join('');

            // Remove possíveis delimitadores markdown que a IA inseriu
            raw = raw
                .replace(/^```json\s*/i, '')
                .replace(/^```\s*/i, '')
                .replace(/```\s*$/i, '')
                .trim();

            // Extrai apenas o objeto JSON (começa em { e termina em })
            const jsonStart = raw.indexOf('{');
            const jsonEnd = raw.lastIndexOf('}');
            if (jsonStart !== -1 && jsonEnd !== -1) {
                raw = raw.slice(jsonStart, jsonEnd + 1);
            }

            const parsed = JSON.parse(raw);
            if (!parsed.questoes || parsed.questoes.length < 5) throw new Error('Questões insuficientes');

            const questoes = parsed.questoes.slice(0, 10);
            // Gera imagens para cada questão baseadas nas keywords sugeridas pela IA
            const imagens = await gerarImagensParaQuestoes(questoes, atividadeState.tema);
            questoes.forEach((q, i) => {
                q._imgUrl = imagens[i].url;
                q._imgCaption = imagens[i].caption || q.imagemCaption || '';
            });

            atividadeState.cache[cacheKey] = questoes;
            renderAtividade(questoes, tipo);

        } catch (err) {
            console.warn('Erro ao chamar API, usando fallback:', err.message);
            const fallback = gerarQuestoesFallback(atividadeState.tituloAE, tipoIdx, atividadeState.tema);
            atividadeState.cache[cacheKey] = fallback;
            renderAtividade(fallback, tipo);
        }
    });

    await _apiQueue;
}

function gerarQuestoesFallback(tema, tipoIdx, temaSlug) {
    const tipo = tiposAtividade[tipoIdx];
    const prefixos = [

        `Em situação de ${tema}:`, `Utilizando ${tema}:`, `Exercício de ${tema}:`, `[Desafio] ${tema}:`
    ];
    const temaImgUrls = [

    ];

    return prefixos.map((p, i) => ({

        opcoes: [
            `A. ${(i + 2) * (i + 1) + (i + 1)}`,
            `B. ${(i + 2) * (i + 1)}`,
            `C. ${i + 3}`,
            `D. ${(i + 2) * (i + 2)}`

    }));
}

function _carregarImagemQuestao(qi) {
    const imgEl = document.getElementById(`qimg_${qi}`);
    if (!imgEl) return;
    const src = imgEl.getAttribute('data-src');
    if (!src) { _onQImgError(qi); return; }

    const tmp = new Image();
    tmp.onload = () => {
        imgEl.src = src;
        imgEl.style.opacity = '1';
        imgEl.classList.add('qi-loaded');
        const skel = document.getElementById(`qimg_skel_${qi}`);
        if (skel) skel.style.opacity = '0';
    };
    tmp.onerror = () => _onQImgError(qi);
    tmp.src = src;
}

function _onQImgError(qi) {
    const wrap = document.getElementById(`qimg_wrap_${qi}`);
    const skel = document.getElementById(`qimg_skel_${qi}`);
    if (skel) {
        skel.innerHTML = `<div class="questao-img-error">📐<br>Conteúdo matemático</div>`;
    }
    if (wrap) {
        wrap.style.minHeight = '64px';
        wrap.style.maxHeight = '64px';
    }
}

function responderQuestao(qi, oi) {
    if (atividadeState.respostas[qi] !== null) return;
    const q = atividadeState.questoes[qi];
    atividadeState.respostas[qi] = oi;
    const isCorreta = (oi === q.correta);
    if (isCorreta) { atividadeState.pontos += 100; addXP(100); }

    const opcoes = document.querySelectorAll(`#opcoes_${qi} .questao-opcao`);
    opcoes.forEach((el, idx) => {
        el.classList.add('disabled');
        if (idx === q.correta) el.classList.add('correta');
        if (idx === oi && !isCorreta) el.classList.add('errada');
    });

    const fb = document.getElementById(`feedback_${qi}`);
    fb.innerHTML = `${isCorreta ? '✅ <strong>Correto!</strong>' : '❌ <strong>Incorreto.</strong>'} ${q.explicacao}`;
    fb.classList.add('visible');

    atividadeState.respondidas++;
    const total = atividadeState.questoes.length;
    document.getElementById('atividadeRespondidas').textContent = `${atividadeState.respondidas}/${total}`;
    document.getElementById('atividadeScorePill').textContent = `${atividadeState.pontos} pts`;
    const fill = document.getElementById('atividadeProgressFill');
    if (fill) fill.style.width = ((atividadeState.respondidas / total) * 100) + '%';

    salvarRespostaAluno(qi, oi, isCorreta, q);

    if (atividadeState.respondidas === total) setTimeout(() => mostrarResultadoAtividade(), 800);
    document.getElementById(`questao_${qi}`).scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}

/* ============================================================
   SALVAR RESPOSTAS (localStorage simulando backend)
   ============================================================ */
function salvarRespostaAluno(qi, resposta, correta, questao) {
    try {
        const chave = `respostas_${atividadeState.codigoAtividade}`;
        const dados = JSON.parse(localStorage.getItem(chave) || '{}');
        const nomeAluno = 'Aluno Demo';
        if (!dados[nomeAluno]) dados[nomeAluno] = { nome: nomeAluno, respostas: [], pontos: 0, habilidades: {} };
        dados[nomeAluno].respostas[qi] = { questao: qi, resposta, correta, enunciado: questao.enunciado };
        if (correta) dados[nomeAluno].pontos += 100;
        const tipo = tiposAtividade[atividadeState.tipoIdx];
        tipo.habilidades.forEach((h, hi) => {
            if (!dados[nomeAluno].habilidades[h]) dados[nomeAluno].habilidades[h] = { tentativas: 0, acertos: 0 };
            if (qi % 5 === hi || qi === hi) {
                dados[nomeAluno].habilidades[h].tentativas++;
                if (correta) dados[nomeAluno].habilidades[h].acertos++;
            }
        });
        localStorage.setItem(chave, JSON.stringify(dados));
    } catch (e) {}
}

function exportarExcel() {
    const tipo = tiposAtividade[atividadeState.tipoIdx];
    const alunos = gerarAlunosSimulados();
    let csv = 'Aluno,Acertos,Aproveitamento (%),' + tipo.habilidades.join(',') + '\n';
    alunos.forEach(a => {
        const row = [a.nome, a.acertos, a.pct, ...tipo.habilidades.map(h => a.habilidades[h] === 'atingida' ? 'Atingida' : a.habilidades[h] === 'parcial' ? 'Parcial' : 'Não Atingida')];
        csv += row.map(v => `"${v}"`).join(',') + '\n';
    });
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = `Dashboard_${atividadeState.codigoAtividade}.csv`;
    a.click(); URL.revokeObjectURL(url);
    showToast('📊 Excel exportado!');
}

    if (arenaState.idx >= arenaState.questions.length) { finishArena(); return; }

function startArenaTimer() {
    if (arenaState.timer) clearInterval(arenaState.timer);
    let sec = 15;
    document.getElementById('arenaFeedback').innerHTML = `⏱ ${sec}s`;
    arenaState.timer = setInterval(() => { sec--; document.getElementById('arenaFeedback').innerHTML = `⏱ ${sec}s`; if (sec <= 0 && !arenaState.answered) { clearInterval(arenaState.timer); answerArena(-1); } }, 1000);
}

function answerArena(sel) {
    if (arenaState.answered) return;
    clearInterval(arenaState.timer);
    const q = arenaState.questions[arenaState.idx];
    const isCorrect = (sel === q.correct);
    if (isCorrect) { arenaState.score += 100; arenaState.corrects++; document.getElementById('arenaScoreVal').innerText = arenaState.score; addXP(100); }
    const fb = isCorrect ? `✅ Correto! ${q.expl || ''}` : `❌ Errado! Resposta: ${q.opts[q.correct]}. ${q.expl || ''}`;
    document.getElementById('arenaFeedback').innerHTML = `<div class="explicacao-box">${fb}</div>`;
    arenaState.answered = true;
    document.querySelectorAll('.arena-opt').forEach((btn, i) => { btn.classList.add('disabled'); if (i === q.correct) btn.classList.add('correct'); if (i === sel && !isCorrect) btn.classList.add('wrong'); });
    document.getElementById('arenaAcertos').innerHTML = `${arenaState.corrects}/${arenaState.questions.length}`;
    arenaState.idx++;
    document.getElementById('arenaNextBtn').style.display = 'block';
    document.getElementById('arenaProgressFill').style.width = ((arenaState.idx / arenaState.questions.length) * 100) + '%';
}

/* ============================================================
   LEITURA DE LINK COMPARTILHADO
   Abre a atividade automaticamente se a URL tiver ?atividade=XXX
   ============================================================ */
function _verificarLinkAtividade() {
    try {
        const params = new URLSearchParams(window.location.search);
        const cod = params.get('atividade');
        if (!cod) return;

        // Extrai dados do código: SERIE-TEMA-TIDX-RANDOM
        const partes = cod.split('-');
        if (partes.length < 3) return;

        const serieMap = { EF6: 'ef6', EF7: 'ef7', EF8: 'ef8', EF9: 'ef9', EM1: 'em1', EM2: 'em2', EM3: 'em3' };
        const temaMap = {
            DECI: 'decimal', RETA: 'reta', OPER: 'operacoes', MULT: 'multiplos',
            GEOM: 'geometria3d', GRAF: 'graficos', INTE: 'inteiros', FRAC: 'fracao',
            POTE: 'potencia', PITA: 'pitagoras', SIST: 'sistema', SEME: 'semelhanca',
            QUAD: 'quadratica', TRIG: 'trigonometria', AFIM: 'afim', EXPO: 'exponencial',
            CICL: 'ciclo', PROG: 'progressao', ANAL: 'analitica'
        };

        const serie = serieMap[partes[0]] || 'ef6';
        const tema = temaMap[partes[1]] || 'geometria3d';
        const tipoIdx = parseInt((partes[2] || 'T0').replace('T', '')) || 0;
        const tituloAE = (nomesSim && nomesSim[tema]) ? nomesSim[tema] : tema;

        // Exibe banner informativo
        showToast(`🔗 Abrindo atividade: ${cod.slice(0, 16)}…`);

        // Abre o modal após um pequeno delay (garante DOM pronto)
        setTimeout(() => {
            openAtividadeModal(tema, tituloAE, serie);
            // Seleciona o tipo correto
            setTimeout(() => {
                const btn = document.querySelector(`.tipo-btn[data-tipo="${tipoIdx}"]`);
                if (btn) selecionarTipo(tipoIdx, btn);
            }, 300);
        }, 500);
    } catch (e) {
        console.warn('Erro ao ler link de atividade:', e);
    }
}

initSelectors();
